# Exemple de POO avec une classe Véhicule
class Vehicule:
    def __init__(self, marque, modele, couleur):
        self.marque = marque
        self.modele = modele
        self.couleur = couleur
        self.kilometrage = 0

    def afficher_description(self):
        return f"{self.couleur} {self.marque} {self.modele}, {self.kilometrage} km"

    def ajouter_kilometrage(self, km):
        if km >= 0:
            self.kilometrage += km
            print(f"{km} km ajoutés. Total : {self.kilometrage} km")
        else:
            print("Kilométrage invalide")

# Exemple de gestion d'exceptions
def lire_fichier_et_traiter(chemin_fichier):
    try:
        with open(chemin_fichier, 'r') as fichier:
            contenu = fichier.read()
            nombres = [int(x) for x in contenu.split(',')]
            return sum(nombres) / len(nombres)
    except FileNotFoundError:
        print(f"Fichier '{chemin_fichier}' non trouvé.")
    except ValueError:
        print("Format de données invalide.")
    except Exception as e:
        print(f"Erreur : {e}")
